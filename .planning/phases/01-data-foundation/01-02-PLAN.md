    This is the shared test fixture that creates an in-memory SQLite database for each test class. Each test class that inherits from DataTestBase gets a fresh, isolated database.

    ```csharp
    using Microsoft.Data.Sqlite;
    using Microsoft.EntityFrameworkCore;
    using BlazorApp2.Data;

    namespace BlazorApp2.Tests;

    public class DataTestBase : IDisposable
    {
        private readonly SqliteConnection _connection;
        protected readonly AppDbContext Context;

        public DataTestBase()
        {
            _connection = new SqliteConnection("DataSource=:memory:");
            _connection.Open();

            var options = new DbContextOptionsBuilder<AppDbContext>()
                .UseSqlite(_connection)
                .Options;

            Context = new AppDbContext(options);
            Context.Database.EnsureCreated();
        }

        public void Dispose()
        {
            Context.Dispose();
            _connection.Dispose();
        }
    }
    ```

    Key implementation notes:
    - Uses `SqliteConnection` with `DataSource=:memory:` -- the connection must stay open for the lifetime of the test, or the in-memory database is destroyed
    - `EnsureCreated()` applies the model (not migrations) to create tables -- this is correct for tests
    - Each test class inherits this and gets `Context` property for database operations
    - `IDisposable` pattern ensures cleanup after each test class
