       ```razor
       @using BlazorApp2.Data.Enums

       <div class="dow-toggle" role="group" aria-label="Days of week">
           @foreach (var day in _days)
           {
               var isActive = Value.HasFlag(day.Flag);
               <button class="dow-chip @(isActive ? "dow-chip--active" : "")"

       @code {
           [Parameter] public DaysOfWeek Value { get; set; }
           [Parameter] public EventCallback<DaysOfWeek> ValueChanged { get; set; }

           private record DayInfo(string Label, string FullName, DaysOfWeek Flag);

           private static readonly DayInfo[] _days = new[]
           {
               new DayInfo("M", "Monday", DaysOfWeek.Monday),
               new DayInfo("T", "Tuesday", DaysOfWeek.Tuesday),
               new DayInfo("W", "Wednesday", DaysOfWeek.Wednesday),
               new DayInfo("T", "Thursday", DaysOfWeek.Thursday),
               new DayInfo("F", "Friday", DaysOfWeek.Friday),
               new DayInfo("S", "Saturday", DaysOfWeek.Saturday),
               new DayInfo("S", "Sunday", DaysOfWeek.Sunday),
           };

           private async Task ToggleDay(DaysOfWeek flag)
           {
               var newValue = Value ^ flag;
               // Prevent deselecting the last active day
               if (newValue == DaysOfWeek.None) return;
               await ValueChanged.InvokeAsync(newValue);
           }
       }
       ```

    3. **Create `Components/Shared/TemplatePicker.razor`**:
       ```razor
       @using BlazorApp2.Data
       @using BlazorApp2.Data.Entities
       @inject IDbContextFactory<AppDbContext> DbFactory

       <div class="template-picker">
           <input class="template-picker__search"
                  type="text"
                  placeholder="Search templates..."
                  @bind="searchText"
                  @bind:event="oninput" />

           <div class="template-picker__list">
               @if (filteredTemplates.Any())
               {
                   @foreach (var template in filteredTemplates)
                   {
                       var isSelected = SelectedTemplateId == template.Id;
                       <div class="template-picker__row @(isSelected ? "template-picker__row--selected" : "")"
                            @onclick="() => SelectTemplate(template)"
                            tabindex="0"
                            role="option"
                            aria-selected="@isSelected">
                           <div class="template-picker__info">
                               <span class="template-picker__name">@template.Name</span>
                               <span class="template-picker__count">@template.Items.Count exercises</span>
                           </div>
                           <span class="template-picker__duration">~@EstimateDuration(template) min</span>
                       </div>
                   }
               }
               else if (allTemplates.Any())
               {
                   <div class="template-picker__empty">No templates match your search</div>
               }
               else
               {
                   <div class="template-picker__empty">
                       No templates yet.
                       <a href="/templates" class="template-picker__link">Create one first.</a>
                   </div>
               }
           </div>
       </div>

       @code {
           [Parameter] public int? SelectedTemplateId { get; set; }
           [Parameter] public EventCallback<WorkoutTemplate> OnTemplateSelected { get; set; }

           private List<WorkoutTemplate> allTemplates = new();
           private string searchText = "";

           private IEnumerable<WorkoutTemplate> filteredTemplates =>
               string.IsNullOrWhiteSpace(searchText)
                   ? allTemplates
                   : allTemplates.Where(t => t.Name.Contains(searchText, StringComparison.OrdinalIgnoreCase));

           protected override async Task OnInitializedAsync()
           {
               using var context = await DbFactory.CreateDbContextAsync();
               allTemplates = await context.WorkoutTemplates
                   .Include(t => t.Items)
                   .OrderBy(t => t.Name)
                   .ToListAsync();
           }

           private async Task SelectTemplate(WorkoutTemplate template)
           {
               await OnTemplateSelected.InvokeAsync(template);
           }

           private int EstimateDuration(WorkoutTemplate template)
           {
               // Simple estimate: count items * 1.5 minutes for strength, actual duration for endurance
               var total = template.Items.Sum(i =>
               {
                   if (i.TargetDurationSeconds.HasValue)
                       return i.TargetDurationSeconds.Value / 60.0;
                   if (i.TargetSets.HasValue)
                       return i.TargetSets.Value * 1.5;
                   return 3.0; // default per exercise
               });
               var rounded = (int)(Math.Ceiling(total / 5.0) * 5);
               return Math.Max(5, rounded);
           }
       }
       ```

       ```csharp
       private async Task HandleSchedule()
       {
           // Validation
           if (!isAdHoc && selectedTemplateId == null) return; // need template
           if (isAdHoc && string.IsNullOrWhiteSpace(adHocName)) return; // need name

           RecurrenceRule? recurrence = null;
           if (isRecurring)
           {
               recurrence = new RecurrenceRule
               {
                   WorkoutTemplateId = isAdHoc ? null : selectedTemplateId,
                   AdHocName = isAdHoc ? adHocName : null,
                   FrequencyType = frequencyType,
                   Interval = frequencyType == FrequencyType.Daily ? interval : 1,
                   DaysOfWeek = frequencyType == FrequencyType.Weekly ? selectedDays : DaysOfWeek.None,
                   StartDate = scheduledDate
               };
           }

           await SchedulingService.ScheduleWorkoutAsync(
               isAdHoc ? null : selectedTemplateId,
               isAdHoc ? adHocName : null,
               scheduledDate,
               recurrence
           );

           await OnScheduled.InvokeAsync();
           await OnClose.InvokeAsync();
       }
       ```
