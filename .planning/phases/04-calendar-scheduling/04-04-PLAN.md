       ```csharp
       private string GetRecurrenceSummary(RecurrenceRule rule)
       {
           if (rule.FrequencyType == FrequencyType.Daily)
               return $"Every {rule.Interval} days";

           var dayNames = new List<string>();
           if (rule.DaysOfWeek.HasFlag(DaysOfWeek.Monday)) dayNames.Add("Monday");
           if (rule.DaysOfWeek.HasFlag(DaysOfWeek.Tuesday)) dayNames.Add("Tuesday");
           if (rule.DaysOfWeek.HasFlag(DaysOfWeek.Wednesday)) dayNames.Add("Wednesday");
           if (rule.DaysOfWeek.HasFlag(DaysOfWeek.Thursday)) dayNames.Add("Thursday");
           if (rule.DaysOfWeek.HasFlag(DaysOfWeek.Friday)) dayNames.Add("Friday");
           if (rule.DaysOfWeek.HasFlag(DaysOfWeek.Saturday)) dayNames.Add("Saturday");
           if (rule.DaysOfWeek.HasFlag(DaysOfWeek.Sunday)) dayNames.Add("Sunday");
           return $"Every {string.Join(", ", dayNames)}";
       }
       ```

       **Helper for target summary:**
       ```csharp
       private string GetTargetSummary(TemplateItem item)
       {
           if (item.TargetSets.HasValue && item.TargetReps.HasValue)
           {
               var weight = item.TargetWeight.HasValue ? $" @ {item.TargetWeight}kg" : "";
               return $"{item.TargetSets}x{item.TargetReps}{weight}";
           }
           var parts = new List<string>();
           if (item.TargetDistance.HasValue) parts.Add($"{item.TargetDistance}km");
           if (item.TargetDurationSeconds.HasValue)
           {
               var ts = TimeSpan.FromSeconds(item.TargetDurationSeconds.Value);
               parts.Add(ts.TotalHours >= 1 ? ts.ToString(@"h\:mm\:ss") : ts.ToString(@"m\:ss"));
           }
           return parts.Any() ? string.Join(" / ", parts) : "";
       }
       ```

       Follow the same pattern as `template-builder.js` -- global namespace, init function, SortableJS `group` config for cross-container drag:

       ```javascript
       window.calendarDrag = {
           sortables: [],

           init: function(dotNetHelper) {
               // Destroy existing instances
               this.sortables.forEach(s => s.destroy());
               this.sortables = [];

               // Find all day cells
               const dayCells = document.querySelectorAll('.day-cell[data-day]');

               dayCells.forEach(cell => {
                   const sortable = Sortable.create(cell, {
                       group: 'calendar-days',
                       animation: 150,
                       ghostClass: 'workout-chip--dragging',
                       dragClass: 'workout-chip--sortable',
                       filter: '.day-add-btn, .day-date, .day-header-mobile',
                       draggable: '.workout-chip',

                       onEnd: function(evt) {
                           const workoutId = parseInt(evt.item.getAttribute('data-workout-id'));
                           const newDay = evt.to.getAttribute('data-day');

                           if (evt.from !== evt.to && workoutId && newDay) {
                               // Revert DOM -- let Blazor re-render
                               evt.from.insertBefore(evt.item, evt.from.children[evt.oldIndex]);

                               dotNetHelper.invokeMethodAsync('OnWorkoutDropped', workoutId, newDay);
                           }
                       }
                   });
                   this.sortables.push(sortable);
               });
           },

           dispose: function() {
               this.sortables.forEach(s => s.destroy());
               this.sortables = [];
           }
       };
       ```
