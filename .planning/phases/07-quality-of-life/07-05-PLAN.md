```csharp
private async Task ExportCsv()
{
    isExportingCsv = true;
    StateHasChanged();
    try
    {
        var strengthBytes = await ExportService.GenerateStrengthCsvAsync(RangeStart, RangeEnd);
        var enduranceBytes = await ExportService.GenerateEnduranceCsvAsync(RangeStart, RangeEnd);

        // Combine: strength CSV + blank line + endurance CSV (if endurance has data beyond header)
        // Simpler: export as single combined CSV with all data, or two separate downloads
        // Per D-09: separate columns for strength and endurance, so export as two files
        // For UX simplicity: export strength data as single file (most common use case)
        // Actually per D-09, there are separate column sets. Export combined:
        // Generate strength CSV and download it
        var fileName = $"workout-data-{DateTime.UtcNow:yyyy-MM-dd}.csv";
        using var stream = new MemoryStream(strengthBytes);
        using var streamRef = new DotNetStreamReference(stream);
        await JS.InvokeVoidAsync("downloadFileFromStream", fileName, streamRef);

        // If endurance data exists, download separately
        if (enduranceBytes.Length > 50) // more than just a header
        {
            var enduranceFileName = $"endurance-data-{DateTime.UtcNow:yyyy-MM-dd}.csv";
            using var endStream = new MemoryStream(enduranceBytes);
            using var endStreamRef = new DotNetStreamReference(endStream);
            await JS.InvokeVoidAsync("downloadFileFromStream", enduranceFileName, endStreamRef);
        }
    }
    catch (Exception)
    {
        // Show error toast -- reuse Toast pattern from existing components
        // For now, the error is silently caught; can add toast notification
    }
    finally
    {
        isExportingCsv = false;
        StateHasChanged();
    }
}
```

```csharp
private OverloadSuggestion? GetSuggestionForExercise(int exerciseId)
{
    if (dismissedSuggestions.Contains(exerciseId)) return null;
    return overloadSuggestions.FirstOrDefault(s => s.ExerciseId == exerciseId);
}

private async Task ApplyOverload(OverloadSuggestion suggestion)
{
    // Update PlannedWeight for all SetLogs of this exercise in current session
    if (workoutLog != null)
    {
        foreach (var setLog in workoutLog.SetLogs.Where(sl => sl.ExerciseId == suggestion.ExerciseId))
        {
            setLog.PlannedWeight = suggestion.SuggestedWeight;
            setLog.ActualWeight = suggestion.SuggestedWeight; // Pre-fill actual too
        }
        // Persist the updated weights
        await SessionService.SaveSetLogAsync(workoutLog.SetLogs.First(sl => sl.ExerciseId == suggestion.ExerciseId));
    }
    dismissedSuggestions.Add(suggestion.ExerciseId);
    // Show toast: "Weight updated to {newWeight}kg"
    // Use existing Toast component pattern
    StateHasChanged();
}

private void DismissOverload(int exerciseId)
{
    dismissedSuggestions.Add(exerciseId);
    StateHasChanged();
}
```
