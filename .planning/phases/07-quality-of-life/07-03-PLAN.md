**Helper Methods:**
- `private string GetWorkoutTypeClass(ScheduledWorkout sw)` -- returns CSS class: "strength" / "endurance" / "mixed" / "adhoc" based on `SchedulingService.DetermineWorkoutType(sw)`
- `private string GetWorkoutTypeLabel(ScheduledWorkout sw)` -- returns "Strength" / "Endurance" / "Mixed" / "Ad Hoc"
- `private string GetWorkoutTypeDotColor(string type)` -- returns CSS variable: type == "Strength" -> "var(--color-strength-text)", "Endurance" -> "var(--color-endurance-text)", "Mixed" -> "var(--color-mixed-text)"
- `private string FormatTargets(TemplateItem item)` -- if item.Exercise is StrengthExercise: return "{TargetSets}x{TargetReps} @ {TargetWeight}kg". If EnduranceExercise: return "{TargetDistance}km / {FormatDuration(TargetDurationSeconds)}". Handle null values gracefully.
- `private string FormatHistoryTargets(HistoryExerciseDetail ex)` -- for last completed view: if strength, format from Sets as "{count}x{reps} @ {weight}kg" using first set's values. If endurance, format from Endurance detail.
- `private int GetVisibleExerciseCount()` -- returns 5 (per UI-SPEC: show first 5, then "+N more")
- `private int GetRemainingExerciseCount(ScheduledWorkout sw)` -- returns total items count - 5 (clamped to 0)

**Action Methods:**

**Home.razor** -- Replace the entire file. Structure:

```razor
@page "/"
@rendermode InteractiveServer

<PageTitle>Home</PageTitle>

<div class="home-page">
    @if (isLoading)
    {
        <!-- Loading state: minimal spinner or skeleton -->
    }
    else if (todaysWorkout != null)
    {
        <!-- STATE 1: Today's Workout (D-01) -->
        <section class="dashboard-section" aria-label="Today's scheduled workout">
            <span class="section-label">TODAY'S WORKOUT</span>
            <div class="dashboard-card">
                <!-- Type dot + Workout name + type label -->
                <div class="card-header">
                    <span class="type-dot" style="background: @GetWorkoutTypeDotColor(...)"></span>
                    <h2 class="workout-name">@todaysWorkout.DisplayName</h2>
                    <span class="type-label type-label--{type}">@GetWorkoutTypeLabel(todaysWorkout)</span>
                </div>

                <!-- Exercise list (first 5 per UI-SPEC) -->
                <ol class="exercise-list">
                    @{
                        var items = todaysWorkout.WorkoutTemplate?.Items?.OrderBy(i => i.Position).ToList() ?? new();
                        var visibleItems = showAllExercises ? items : items.Take(5).ToList();
                    }
                    @foreach (var (item, index) in visibleItems.Select((it, i) => (it, i)))
                    {
                        <li class="exercise-list__item">
                            <span class="exercise-pos">@(index + 1).</span>
                            <span class="exercise-name">@item.Exercise.Name</span>
                            <span class="exercise-targets">@FormatTargets(item)</span>
                        </li>
                    }
                </ol>

                @if (items.Count > 5 && !showAllExercises)
                {
                    <button class="show-more-btn" @onclick="ToggleExerciseList">
                        +@(items.Count - 5) more exercises
                    </button>
                }
                @if (showAllExercises && items.Count > 5)
                {
                    <button class="show-more-btn" @onclick="ToggleExerciseList">
                        Show less
                    </button>
                }

                <!-- Start Session CTA -->
                <button class="btn-start-session" @onclick="StartSession">
                    <svg><!-- 16x16 play icon SVG --></svg>
                    Start Session
                </button>
            </div>
        </section>

        <!-- UP NEXT section (D-02) -->
        @if (tomorrowsWorkout != null)
        {
            <div class="up-next">
                <span class="section-label">UP NEXT</span>
                <span class="up-next__date"> -- Tomorrow</span>
                <div class="up-next__workout">
                    <span class="type-dot type-dot--sm" style="background: @GetWorkoutTypeDotColor(...)"></span>
                    <span class="up-next__name">@tomorrowsWorkout.DisplayName</span>
                </div>
            </div>
        }
    }
    else if (lastCompleted != null)
    {
        <!-- STATE 2: Last Completed Workout (D-03) -->
        <section class="dashboard-section" aria-label="Last completed workout">
            <span class="section-label">LAST COMPLETED</span>
            <div class="dashboard-card">
                <div class="card-header">
                    <span class="type-dot" style="background: @GetWorkoutTypeDotColor(...)"></span>
                    <h2 class="workout-name">@lastCompleted.WorkoutName</h2>
                    <span class="completed-date">@lastCompleted.CompletedAt.ToString("MMM d, yyyy")</span>
                </div>

                <!-- Exercise list with ACTUAL logged values (not template targets) per D-03 -->
                <ol class="exercise-list">
                    @foreach (var (ex, index) in lastCompleted.Exercises.Take(showAllExercises ? int.MaxValue : 5).Select((e, i) => (e, i)))
                    {
                        <li class="exercise-list__item">
                            <span class="exercise-pos">@(index + 1).</span>
                            <span class="exercise-name">@ex.ExerciseName</span>
                            <span class="exercise-targets">@FormatHistoryTargets(ex)</span>
                        </li>
                    }
                </ol>

                @if (lastCompleted.Exercises.Count > 5 && !showAllExercises)
                {
                    <button class="show-more-btn" @onclick="ToggleExerciseList">
                        +@(lastCompleted.Exercises.Count - 5) more exercises
                    </button>
                }

                <!-- Repeat Workout CTA -->
                <button class="btn-start-session" @onclick="RepeatWorkout" disabled="@isRepeating">
                    <svg><!-- 16x16 repeat/refresh icon SVG --></svg>
                    @(isRepeating ? "Scheduling..." : "Repeat Workout")
                </button>
            </div>
        </section>

        <div class="or-divider">or</div>
        <button class="btn-ghost" @onclick="NavigateToTemplates">Browse Templates</button>
    }
    else if (!hasAnyTemplates)
    {
        <!-- STATE 3: Empty (no workouts ever) -->
        <div class="empty-state">
            <h2 class="empty-state__heading">Welcome to Workout Planner</h2>
            <p class="empty-state__body">Build your first workout template to get started.</p>
            <div class="empty-state__actions">
                <button class="btn-ghost" @onclick="NavigateToNewTemplate">Create Template</button>
                <button class="btn-ghost" @onclick="NavigateToExercises">Browse Exercises</button>
            </div>
        </div>
    }
    else
    {
        <!-- STATE 4: Has templates but nothing today and no completed sessions -->
        <div class="empty-state">
            <h2 class="empty-state__heading">Nothing scheduled for today</h2>
            <p class="empty-state__body">Schedule a workout from the calendar or start fresh from a template.</p>
            <div class="empty-state__actions">
                <button class="btn-ghost" @onclick="NavigateToCalendar">Go to Calendar</button>
                <button class="btn-ghost" @onclick="NavigateToTemplates">Browse Templates</button>
            </div>
        </div>
    }
</div>
```

Provide inline SVGs for play icon (16x16 triangle pointing right) and repeat icon (16x16 circular arrows). Use simple path-based SVGs matching existing project convention.
