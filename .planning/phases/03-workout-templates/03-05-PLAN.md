   ```javascript
   // Template builder drag-and-drop interop module
   // Uses SortableJS (loaded globally via script tag in App.razor)

   let sortableInstance = null;

   export function initSortable(dotNetRef, containerSelector) {
       // Destroy previous instance if exists
       if (sortableInstance) {
           sortableInstance.destroy();
           sortableInstance = null;
       }

       const el = document.querySelector(containerSelector);
       if (!el) {
           console.warn('SortableJS: container not found:', containerSelector);
           return;
       }

       sortableInstance = new Sortable(el, {
           handle: '.drag-handle',
           animation: 150,
           ghostClass: 'drag-placeholder',
           chosenClass: 'drag-chosen',
           dragClass: 'drag-active',
           fallbackOnBody: true,
           swapThreshold: 0.65,
           onEnd: function (evt) {
               // CRITICAL: Revert DOM change -- let Blazor re-render
               // SortableJS moves the DOM element, but Blazor owns the DOM.
               // We must put it back and let .NET update the list.
               const item = evt.item;
               const from = evt.from;

               // Remove the moved item from its new position
               item.remove();

               // Re-insert at original position
               if (evt.oldIndex < from.children.length) {
                   from.insertBefore(item, from.children[evt.oldIndex]);
               } else {
                   from.appendChild(item);
               }

               // Notify .NET of the reorder
               dotNetRef.invokeMethodAsync('OnItemReordered', evt.oldIndex, evt.newIndex);
           }
       });
   }

   export function refreshSortable(containerSelector) {
       // Call after Blazor re-renders the list (add/remove exercises)
       if (sortableInstance) {
           // SortableJS may lose track of new elements
           // Easiest fix: just ensure the instance is still bound
           const el = document.querySelector(containerSelector);
           if (el && sortableInstance.el !== el) {
               // Container element was replaced by Blazor
               sortableInstance.destroy();
               sortableInstance = null;
               // Will be re-initialized on next call to initSortable
           }
       }
   }

   export function destroySortable() {
       if (sortableInstance) {
           sortableInstance.destroy();
           sortableInstance = null;
       }
   }
   ```

   protected override async Task OnAfterRenderAsync(bool firstRender)
   {
       if (firstRender)
       {
           _jsModule = await JS.InvokeAsync<IJSObjectReference>(
               "import", "./js/template-builder.js");
           _dotNetRef = DotNetObjectReference.Create(this);
       }

       // Initialize or refresh sortable when items exist
       if (State.Items.Count > 0 && _jsModule != null && _dotNetRef != null)
       {
           if (!_sortableInitialized || _lastItemCount != State.Items.Count)
           {
               await _jsModule.InvokeVoidAsync("initSortable", _dotNetRef, ".exercise-list-sortable");
               _sortableInitialized = true;
               _lastItemCount = State.Items.Count;
           }
       }
   }

   **JSInvokable callback for drag reorder:**
   ```csharp
   [JSInvokable]
   public void OnItemReordered(int oldIndex, int newIndex)
   {
       if (oldIndex == newIndex) return;

       State.PushUndo();

       // Get the flat list of items in current display order
       var orderedItems = State.Items.OrderBy(i => i.Position).ToList();

       if (oldIndex < 0 || oldIndex >= orderedItems.Count || newIndex < 0 || newIndex >= orderedItems.Count)
           return;

       // Move item
       var item = orderedItems[oldIndex];
       orderedItems.RemoveAt(oldIndex);
       orderedItems.Insert(newIndex, item);

       // Recompact positions
       for (int i = 0; i < orderedItems.Count; i++)
           orderedItems[i].Position = i;

       StateHasChanged();
   }
   ```

   **Dispose SortableJS on page leave:**
   ```csharp
   public async ValueTask DisposeAsync()
   {
       if (_jsModule != null)
       {
           try
           {
               await _jsModule.InvokeVoidAsync("destroySortable");
               await _jsModule.DisposeAsync();
           }
           catch (JSDisconnectedException) { }
       }
       _dotNetRef?.Dispose();
   }
   ```

   private void HandleKeyDown(KeyboardEventArgs e)
   {
       if (e.CtrlKey && e.Key == "z" && !e.ShiftKey)
       {
           Undo();
       }
       else if (e.CtrlKey && e.Key == "z" && e.ShiftKey)

       {
           Redo();
       }
   }
